36 void DependencyGraph::insert(const DependencyNodePtr &newNode)
37 {
38  nodes_.push_back(newNode);
39 
40  // find the modal node groups that have a shared variable with newNode.
41  std::list<std::list<DependencyGroup>::iterator > dependencies;
42  for(auto it=groups_.begin(); it!=groups_.end(); ++it) {
43  if(has_intersection(it->variables_, newNode->variables())) {
44  dependencies.push_back(it);
45  }
46  }
47 
48  DependencyGroup *newNodeGroup = nullptr;
49  if(dependencies.empty()) {
50  // the literal does not have a shared variable with an existing group.
51  // create a new one with just the literal as a member.
52  auto &newGroup = groups_.emplace_back();
53  newNodeGroup = &newGroup;
54  }
55  else if(dependencies.size()==1) {
56  // add to group
57  auto &group = *dependencies.front();
58  newNodeGroup = &group;
59  }
60  else {
61  // merge multiple groups into one
62  auto &newGroup = groups_.emplace_back();
63  for(auto groupIterator : dependencies) {
64  newGroup += *groupIterator;
65  groups_.erase(groupIterator);
66  }
67  newNodeGroup = &newGroup;
68  }
69  // update neighbor relation based on shared variables with other nodes
70  for(auto &x : newNodeGroup->member_) {
71  if(has_intersection(x->variables(), newNode->variables())) {
72  x->neighbors_.push_back(newNode);
73  newNode->neighbors_.push_back(x);
74  }
75  }
76  // finally add the new node to the group
77  newNodeGroup->member_.push_back(newNode);
78  newNodeGroup->variables_.insert(newNode->variables().begin(), newNode->variables().end());
79 }
