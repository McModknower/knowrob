21 bool InterfaceUtils::assertStatements(KnowledgeBase &kb_, const std::vector<FormulaPtr> &args) {
22  std::vector<FramedTriplePtr> data(args.size());
23  std::vector<FramedTriplePatternPtr> buf(args.size());
24  uint32_t dataIndex = 0;
25 
26  for (auto &phi: args) {
27  const QueryTree qt(phi);
28  if (qt.numPaths() > 1) {
29  throw QueryError("Disjunctions are not allowed in assertions. "
30  "Appears in statement {}.", *phi);
31  } else if (qt.numPaths() == 0) {
32  throw QueryError("Invalid assertion: '{}'", *phi);
33  }
34  for (auto &psi: qt.begin()->nodes()) {
35  switch (psi->type()) {
36  case knowrob::FormulaType::PREDICATE:
37  buf[dataIndex] = std::make_shared<FramedTriplePattern>(
38  std::static_pointer_cast<Predicate>(psi), false);
39  data[dataIndex].ptr = new FramedTripleCopy();
40  data[dataIndex].owned = true;
41  buf[dataIndex]->instantiateInto(*data[dataIndex].ptr);
42  dataIndex += 1;
43  break;
44  default:
45  throw QueryError("Invalid assertion: '{}'", *phi);
46  }
47  }
48  }
49  if (kb_.insertAll(data)) {
50  std::cout << "success, " << dataIndex << " statement(s) were asserted." << "\n";
51  return true;
52  } else {
53  std::cout << "assertion failed." << "\n";
54  return false;
55  }
56 }

60  FormulaPtr phi) {
61  FormulaPtr mFormula = std::move(phi);
62 
63  // Retrieve epistemicOperator and check if it is "BELIEF"
64  auto epistemicOperator = boost::any_cast<int>(options.at("epistemicOperator"));
65  if (epistemicOperator == 1) {
66  // Retrieve aboutAgentIRI and confidence
67  auto aboutAgentIRI = boost::any_cast<std::string>(options.at("aboutAgentIRI"));
68  auto confidence = boost::any_cast<double>(options.at("confidence"));
69  if (!aboutAgentIRI.empty()) {
70  if (confidence != 1.0) {
71  mFormula = std::make_shared<ModalFormula>(
72  modals::B(aboutAgentIRI, confidence), mFormula);
73  } else {
74  mFormula = std::make_shared<ModalFormula>(
75  modals::B(aboutAgentIRI), mFormula);
76  }
77  }
78  } else if (epistemicOperator == 0) {
79  // Retrieve aboutAgentIRI
80  auto aboutAgentIRI = boost::any_cast<std::string>(options.at("aboutAgentIRI"));
81  if (!aboutAgentIRI.empty()) {
82  mFormula = std::make_shared<ModalFormula>(
83  modals::K(aboutAgentIRI), mFormula);
84  }
85  }
86  // Retrieve temporalOperator
87  auto temporalOperator = boost::any_cast<int>(options.at("temporalOperator"));
88 
89  // Retrieve minPastTimestamp and maxPastTimestamp
90  auto minPastTimestamp = boost::any_cast<double>(options.at("minPastTimestamp"));
91  auto maxPastTimestamp = boost::any_cast<double>(options.at("maxPastTimestamp"));
92 
93  auto minPastTimePoint =
94  minPastTimestamp != -1 ? std::optional<TimePoint>(knowrob::time::fromSeconds(minPastTimestamp))
95  : std::nullopt;
96  auto maxPastTimePoint =
97  maxPastTimestamp != -1 ? std::optional<TimePoint>(knowrob::time::fromSeconds(maxPastTimestamp))
98  : std::nullopt;
99 
100  if (temporalOperator == 2) {
101  if (minPastTimestamp != -1 || maxPastTimestamp != -1) {
102  if (minPastTimestamp == -1) {
103  mFormula = std::make_shared<ModalFormula>(
104  modals::P(TimeInterval(std::nullopt,
105  maxPastTimePoint)), mFormula);
106  } else if (maxPastTimestamp == -1) {
107  mFormula = std::make_shared<ModalFormula>(
108  modals::P(TimeInterval(minPastTimePoint,
109  std::nullopt)), mFormula);
110  } else {
111  mFormula = std::make_shared<ModalFormula>(
112  modals::P(TimeInterval(minPastTimePoint,
113  maxPastTimePoint)), mFormula);
114  }
115  } else {
116  mFormula = std::make_shared<ModalFormula>(
117  modals::P(), mFormula);
118  }
119  } else if (temporalOperator == 1) {
120  if (minPastTimestamp != -1 || maxPastTimestamp != -1) {
121  if (minPastTimestamp == -1) {
122  mFormula = std::make_shared<ModalFormula>(
123  modals::H(TimeInterval(std::nullopt,
124  maxPastTimePoint)), mFormula);
125  } else if (maxPastTimestamp == -1) {
