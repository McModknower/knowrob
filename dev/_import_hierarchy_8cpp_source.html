69 void ImportHierarchy::addDirectImport(std::string_view importerGraphName, std::string_view importedGraphName) {
70  auto &g_importer = getCurrentGraph(importerGraphName);
71  auto &g_imported = getCurrentGraph(importedGraphName);
72 
73  // if a system graph imports a session graph, remove the session graph from the session
74  // before adding to the system. And if a session graph imports a system graph, ignore the import.
75  if (isSystemOrigin(g_importer) && isSessionOrigin(g_imported)) {
76  removeCurrentGraph(importedGraphName);
77  } else if (isSessionOrigin(g_importer) && isSystemOrigin(g_imported)) {
78  KB_WARN("Ignoring session graph \"{}\" import of system graph \"{}\".", importerGraphName, importedGraphName);
79  return;
80  }
81 
82  // g_importer.directlyImports += [g_imported]
83  auto pair = g_importer.directImports_.insert(&g_imported);
84  if (!pair.second) return;
85  KB_DEBUG("Graph \"{}\" imports \"{}\".", importerGraphName, importedGraphName);
86  // g_importer.imports += g_imported.imports + [gb]
87  g_importer.imports_.insert(g_imported.imports_.begin(), g_imported.imports_.end());
88  g_importer.imports_.insert(&g_imported);
89  // for every graph gx that imports g_importer:
90  // gx.imports += (g_imported.imports + [g_imported])
91  for (auto &it: graphs_) {
92  if (it.second->imports_.count(&g_importer) > 0) {
93  it.second->imports_.insert(g_imported.imports_.begin(), g_imported.imports_.end());
94  it.second->imports_.insert(&g_imported);
95  }
96  }
97 }

99 void ImportHierarchy::removeCurrentGraph(std::string_view graphName) {
100  auto it = graphs_.find(graphName);
101  if (it == graphs_.end()) return;
102  auto &ga = *it->second;
103 
104  // for every graph gx that directly imports ga
105  for (auto &jt: graphs_) {
106  auto &gx = *jt.second;
107  if (gx.directImports_.count(&ga) > 0) {
108  // gx.directlyImports -= [ga]
109  gx.directImports_.erase(&ga);
110  gx.imports_.erase(&ga);
111  // gx.directlyImports += [ga.directImports]
112  gx.directImports_.insert(ga.directImports_.begin(), ga.directImports_.end());
113  }
114  }
115  // for every graph gy that imports ga:
116  for (auto &kt: graphs_) {
117  auto &gy = *kt.second;
118  if (gy.imports_.count(&ga) > 0) {
119  // gy.imports -= [ga]
120  gy.imports_.erase(&ga);
121  }
122  }
123 
124  // finally delete the node
125  graphs_.erase(it);
126 }
