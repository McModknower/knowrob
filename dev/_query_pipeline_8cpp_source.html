44 QueryPipeline::QueryPipeline(KnowledgeBase *kb, const FormulaPtr &phi, const QueryContextPtr &ctx) {
45  auto outStream = std::make_shared<TokenBuffer>();
46  addStage(outStream);
47 
48  // decompose input formula into parts that are considered in disjunction,
49  // and thus can be evaluated in parallel.
50  QueryTree qt(phi);
51  for (auto &path: qt) {
52  // each node in a path is either a predicate, a negated predicate,
53  // a modal formula, or the negation of a modal formula.
54 
55  std::vector<FramedTriplePatternPtr> posLiterals, negLiterals;
56  std::vector<std::shared_ptr<ModalFormula>> posModals, negModals;
57 
58  for (auto &node: path.nodes()) {
59  switch (node->type()) {
60  case FormulaType::PREDICATE: {
61  auto pat = std::make_shared<FramedTriplePattern>(
62  std::static_pointer_cast<Predicate>(node), false);
63  pat->setTripleFrame(ctx->selector);
64  posLiterals.push_back(pat);
65  break;
66  }
67 
68  case FormulaType::MODAL:
69  posModals.push_back(std::static_pointer_cast<ModalFormula>(node));
70  break;
71 
72  case FormulaType::NEGATION: {
73  auto negation = (Negation *) node.get();
74  auto negated = negation->negatedFormula();
75  switch (negated->type()) {
76  case FormulaType::PREDICATE: {
77  auto pat = std::make_shared<FramedTriplePattern>(
78  std::static_pointer_cast<Predicate>(negated), true);
79  pat->setTripleFrame(ctx->selector);
80  negLiterals.push_back(pat);
81  break;
82  }
83  case FormulaType::MODAL:
84  negModals.push_back(std::static_pointer_cast<ModalFormula>(negated));
85  break;
86  default:
87  throw QueryError("Unexpected negated formula type {} in QueryTree.", (int) negated->type());
88  }
89  break;
90  }
91  default:
92  throw QueryError("Unexpected formula type {} in QueryTree.", (int) node->type());
93  }
94  }
95 
96  std::shared_ptr<TokenBroadcaster> lastStage;
97  std::shared_ptr<TokenBuffer> firstBuffer;
98 
99  // first evaluate positive literals if any
100  // note that the first stage is buffered, so that the next stage can be added to the pipeline
101  // and only after stopping the buffering messages will be forwarded to the next stage.
102  if (posLiterals.empty()) {
103  // if there are none, we still need to indicate begin and end of stream for the rest of the pipeline.
104  // so we just push `bos` (an empty substitution) followed by `eos` and feed these messages to the next stage.
105  firstBuffer = std::make_shared<TokenBuffer>();
106  lastStage = firstBuffer;
107  auto channel = TokenStream::Channel::create(lastStage);
108  channel->push(GenericYes());
109  channel->push(EndOfEvaluation::get());
110  addStage(lastStage);
111  } else {
112  auto pathQuery = std::make_shared<GraphPathQuery>(posLiterals, ctx);
113 
114  auto subPipeline = std::make_shared<QueryPipeline>(kb, pathQuery);
115  firstBuffer = std::make_shared<AnswerBuffer_WithReference>(subPipeline);
116  *subPipeline >> firstBuffer;
117  subPipeline->stopBuffering();
118 
119  lastStage = firstBuffer;
120  addStage(lastStage);
121  }
122 
123  // --------------------------------------
124  // Evaluate all positive modals in sequence.
125  // TODO: compute dependency between modals, evaluate independent modals in parallel.
126  // they could also be independent in evaluation context only, we could check which variables receive
127  // grounding already before in posLiteral query!
128  // this is effectively what is done in submitQuery(pathQuery)
129  // --------------------------------------
130  for (auto &posModal: posModals) {
131  // TODO: improve interaction between kb and modal stage.
132  // use of this pointer not nice.
133  // maybe a better way would be having another class generating pipelines with the ability
134  // to create reference pointer on a KnowledgeBase, or use a weak ptr here
135  auto modalStage = std::make_shared<ModalStage>(kb, posModal, ctx);
136  modalStage->selfWeakRef_ = modalStage;
137  lastStage >> modalStage;
138  lastStage = modalStage;
139  addStage(lastStage);
140  }
141 
142  // --------------------------------------
143  // Evaluate all negative literals in parallel.
144  // --------------------------------------
145  if (!negLiterals.empty()) {
146  // run a dedicated stage where negated literals can be evaluated in parallel
147  auto negLiteralStage = std::make_shared<PredicateNegationStage>(
148  kb, ctx, negLiterals);
149  lastStage >> negLiteralStage;
150  lastStage = negLiteralStage;
151  addStage(lastStage);
152  }
153 
154  // --------------------------------------
155  // Evaluate all negative modals in parallel.
156  // --------------------------------------
157  if (!negModals.empty()) {
158  // run a dedicated stage where negated modals can be evaluated in parallel
159  auto negModalStage = std::make_shared<ModalNegationStage>(
160  kb, ctx, negModals);
161  lastStage >> negModalStage;
162  lastStage = negModalStage;
163  addStage(lastStage);
164  }
165 
166  lastStage >> outStream;
167  firstBuffer->stopBuffering();
168  addStage(lastStage);
169  }
170  // At this point outStream could already contain solutions, but these are buffered
171  // such that they won't be lost during pipeline creation.
172 
173  // if there were multiple paths, consolidate answers from them.
174  // e.g. if one yields no and the other true, the no should be ignored.
175  if (qt.numPaths() > 1) {
176  auto consolidator = std::make_shared<DisjunctiveBroadcaster>();
177  outStream >> consolidator;
178  finalStage_ = consolidator;
179  } else {
180  finalStage_ = outStream;
181  }
182  bufferStage_ = outStream;
183 }

185 QueryPipeline::QueryPipeline(KnowledgeBase *kb, const GraphPathQueryPtr &graphQuery) {
186  auto &allLiterals = graphQuery->path();
187 
188  // --------------------------------------
189  // split input literals into positive and negative literals.
190  // negative literals are evaluated in parallel after all positive literals.
191  // --------------------------------------
192  std::vector<FramedTriplePatternPtr> positiveLiterals, negativeLiterals;
193  for (auto &l: allLiterals) {
194  if (l->isNegated()) negativeLiterals.push_back(l);
195  else positiveLiterals.push_back(l);
196  }
197 
198  // --------------------------------------
199  // split positive literals into edb-only and computable.
200  // also associate list of reasoner to computable literals.
201  // --------------------------------------
202  std::vector<FramedTriplePatternPtr> edbOnlyLiterals;
203  std::vector<RDFComputablePtr> computableLiterals;
204  for (auto &l: positiveLiterals) {
205  auto property = l->propertyTerm();
206  if (!property->isAtom()) {
207  KB_WARN("Variable predicate in query not supported.");
208  continue;
209  }
210  auto property_a = std::static_pointer_cast<Atomic>(property);
211  auto l_reasoner = kb->reasonerManager()->getReasonerForRelation(
212  PredicateIndicator(property_a->stringForm(), 2));
213  if (l_reasoner.empty()) {
214  edbOnlyLiterals.push_back(l);
215  auto l_p = l->propertyTerm();
216  if (l_p && l_p->termType() == TermType::ATOMIC &&
217  !isMaterializedInEDB(kb, std::static_pointer_cast<Atomic>(l_p)->stringForm())) {
218  // generate a "don't know" message and return.
219  auto out = std::make_shared<TokenBuffer>();
220  auto channel = TokenStream::Channel::create(out);
221  auto dontKnow = std::make_shared<AnswerDontKnow>();
222  KB_WARN("Predicate {} is neither materialized in EDB nor defined by a reasoner.", *l->predicate());
223  channel->push(dontKnow);
224  channel->push(EndOfEvaluation::get());
225  finalStage_ = out;
226  bufferStage_ = out;
227  return;
228  }
229  } else {
230  computableLiterals.push_back(std::make_shared<RDFComputable>(*l, l_reasoner));
231  }
232  }
233 
234  // --------------------------------------
235  // sort positive literals.
236  // --------------------------------------
237  std::sort(edbOnlyLiterals.begin(), edbOnlyLiterals.end(), EDBComparator(kb->vocabulary()));
238 
239  // --------------------------------------
240  // run EDB query with all edb-only literals.
241  // --------------------------------------
242  std::shared_ptr<TokenBuffer> edbOut;
243  if (edbOnlyLiterals.empty()) {
244  edbOut = std::make_shared<TokenBuffer>();
245  auto channel = TokenStream::Channel::create(edbOut);
246  channel->push(GenericYes());
247  channel->push(EndOfEvaluation::get());
248  } else {
249  auto edb = kb->getBackendForQuery();
250  edbOut = kb->edb()->getAnswerCursor(edb,
251  std::make_shared<GraphPathQuery>(edbOnlyLiterals, graphQuery->ctx()));
252  }
253  addStage(edbOut);
254 
255  // --------------------------------------
256  // handle positive IDB literals.
257  // --------------------------------------
258  std::shared_ptr<TokenBroadcaster> idbOut;
259  if (computableLiterals.empty()) {
260  idbOut = edbOut;
261  } else {
262  idbOut = std::make_shared<TokenBroadcaster>();
263  addStage(idbOut);
264  // --------------------------------------
265  // Compute dependency groups of computable literals.
266  // --------------------------------------
267  DependencyGraph dg;
268  dg.insert(computableLiterals.begin(), computableLiterals.end());
269 
270  // --------------------------------------
271  // Construct a pipeline for each dependency group.
272  // --------------------------------------
273  if (dg.numGroups() == 1) {
274  auto &literalGroup = *dg.begin();
275  createComputationPipeline(
276  kb,
277  createComputationSequence(kb, literalGroup.member_),
278  edbOut,
279  idbOut,
280  graphQuery->ctx());
281  } else {
282  // there are multiple dependency groups. They can be evaluated in parallel.
283 
284  // combines sub-answers computed in different parallel steps
285  auto answerCombiner = std::make_shared<ConjunctiveBroadcaster>();
286  // create a parallel step for each dependency group
287  for (auto &literalGroup: dg) {
288  // --------------------------------------
289  // Construct a pipeline for each dependency group.
290  // --------------------------------------
291  createComputationPipeline(
292  kb,
293  createComputationSequence(kb, literalGroup.member_),
294  edbOut,
295  answerCombiner,
296  graphQuery->ctx());
297  }
298  answerCombiner >> idbOut;
299  addStage(answerCombiner);
300  }
301  }
302 
303  // --------------------------------------
304  // Evaluate all negative literals in parallel.
305  // --------------------------------------
306  if (!negativeLiterals.empty()) {
307  // run a dedicated stage where negated literals can be evaluated in parallel
308  auto negStage = std::make_shared<PredicateNegationStage>(
309  kb, graphQuery->ctx(), negativeLiterals);
310  idbOut >> negStage;
311  finalStage_ = negStage;
312  } else {
313  finalStage_ = idbOut;
314  }
315  bufferStage_ = edbOut;
316 }

338  // used to sort dependency nodes in a priority queue.
339  // the nodes are considered to be dependent on each other through free variables.
340  // the priority value is used to determine which nodes should be evaluated first.
341  struct DependencyNodeComparator {
342  bool operator()(const DependencyNodePtr &a, const DependencyNodePtr &b) const {
343  // prefer node with fewer variables
344  if (a->numVariables() != b->numVariables()) {
345  return a->numVariables() > b->numVariables();
346  }
347  // prefer node with fewer neighbors
348  if (a->numNeighbors() != b->numNeighbors()) {
349  return a->numNeighbors() > b->numNeighbors();
350  }
351  return a < b;
352  }
353  };

355  struct DependencyNodeQueue {
356  const DependencyNodePtr node_;
357  std::priority_queue<DependencyNodePtr, std::vector<DependencyNodePtr>, DependencyNodeComparator> neighbors_;

359  explicit DependencyNodeQueue(const DependencyNodePtr &node) : node_(node) {
360  // add all nodes to a priority queue
361  for (auto &neighbor: node->neighbors()) {
362  neighbors_.push(neighbor);
363  }
364  }

368 std::vector<RDFComputablePtr> QueryPipeline::createComputationSequence(
369  KnowledgeBase *kb,
370  const std::list<DependencyNodePtr> &dependencyGroup) {
371  // Pick a node to start with.
372  auto comparator = IDBComparator(kb->vocabulary());
373  DependencyNodePtr first;
374  RDFComputablePtr firstComputable;
375  for (auto &n: dependencyGroup) {
376  auto computable_n =
377  std::static_pointer_cast<RDFComputable>(n->literal());
378  if (!first || comparator(firstComputable, computable_n)) {
379  first = n;
380  firstComputable = computable_n;
381  }
382  }
383 
384  // remember visited nodes, needed for circular dependencies
385  // all nodes added to the queue should also be added to this set.
386  std::set<DependencyNode *> visited;
387  visited.insert(first.get());
388 
389  std::vector<RDFComputablePtr> sequence;
390  sequence.push_back(firstComputable);
391 
392  // start with a FIFO queue only containing first node
393  std::deque<std::shared_ptr<DependencyNodeQueue>> queue;
394  auto qn0 = std::make_shared<DependencyNodeQueue>(first);
395  queue.push_front(qn0);
396 
397  // loop until queue is empty and process exactly one successor of
398  // the top element in the FIFO in each step. If an element has no
399  // more successors, it can be removed from queue.
400  // Each successor creates an additional node added to the top of the FIFO.
401  while (!queue.empty()) {
402  auto front = queue.front();
403 
404  // get top successor node that has not been visited yet
405  DependencyNodePtr topNext;
406  while (!front->neighbors_.empty()) {
407  auto topNeighbor = front->neighbors_.top();
408  front->neighbors_.pop();
409 
410  if (visited.count(topNeighbor.get()) == 0) {
411  topNext = topNeighbor;
412  break;
413  }
414  }
415  // pop element from queue if all neighbors were processed
416  if (front->neighbors_.empty()) {
417  queue.pop_front();
418  }
419 
420  if (topNext) {
421  // push a new node onto FIFO
422  auto qn_next = std::make_shared<DependencyNodeQueue>(topNext);
423  queue.push_front(qn_next);
424  sequence.push_back(std::static_pointer_cast<RDFComputable>(topNext->literal()));
425  visited.insert(topNext.get());
426  }
427  }
428 
429  return sequence;
430 }

432 void QueryPipeline::createComputationPipeline(
433  KnowledgeBase *kb,
434  const std::vector<RDFComputablePtr> &computableLiterals,
435  const std::shared_ptr<TokenBroadcaster> &pipelineInput,
436  const std::shared_ptr<TokenBroadcaster> &pipelineOutput,
437  const QueryContextPtr &ctx) {
438  // This function generates a query pipeline for literals that can be computed
439  // (EDB-only literals are processed separately). The literals are part of one dependency group.
440  // They are sorted, and also evaluated in this order. For each computable literal there is at
441  // least one reasoner that can compute the literal. However, instances of the literal may also
442  // occur in the EDB. Hence, computation results must be combined with results of an EDB query
443  // for each literal.
444 
445  auto lastOut = pipelineInput;
446 
447  for (auto &lit: computableLiterals) {
448  auto stepInput = lastOut;
449  auto stepOutput = std::make_shared<TokenBroadcaster>();
450  addStage(stepOutput);
451 
452  // --------------------------------------
453  // Construct a pipeline that grounds the literal in the EDB.
454  // But only add an EDB stage if the predicate was materialized in EDB before,
455  // or if the predicate is a variable.
456  // --------------------------------------
457  bool isEDBStageNeeded = true;
458  if (lit->propertyTerm() && lit->propertyTerm()->termType() == TermType::ATOMIC) {
459  isEDBStageNeeded = isMaterializedInEDB(kb,
460  std::static_pointer_cast<Atomic>(lit->propertyTerm())->stringForm());
461  }
462  if (isEDBStageNeeded) {
463  auto edb = kb->getBackendForQuery();
464  auto edbStage = std::make_shared<TypedQueryStage<FramedTriplePattern>>(
465  ctx,
466  lit,
467  [&](const FramedTriplePatternPtr &q) {
468  return kb->edb()->getAnswerCursor(edb, std::make_shared<GraphPathQuery>(q, ctx));
469  });
470  edbStage->selfWeakRef_ = edbStage;
471  stepInput >> edbStage;
472  edbStage >> stepOutput;
473  addStage(edbStage);
474  }
475 
476  // --------------------------------------
477  // Construct a pipeline that grounds the literal in the IDB.
478  // To this end add an IDB stage for each reasoner that defines the literal.
479  // --------------------------------------
480  for (auto &r: lit->reasonerList()) {
481  auto idbStage = std::make_shared<TypedQueryStage<FramedTriplePattern>>(
482  ctx, lit,
483  [&r, &ctx](const FramedTriplePatternPtr &q) {
484  return r->submitQuery(q, ctx);
485  });
486  idbStage->selfWeakRef_ = idbStage;
487  stepInput >> idbStage;
488  idbStage >> stepOutput;
489  addStage(idbStage);
490  // TODO: what about the materialization of the predicate in EDB?
491  }
492 
493  // add a stage that consolidates the results of the EDB and IDB stages.
494  // in particular the case needs to be handled where none of the stages return
495  // 'true'. Also print a warning if two stages disagree but state they are confident.
496  auto consolidator = std::make_shared<DisjunctiveBroadcaster>();
497  stepOutput >> consolidator;
498  // TODO: are all these addStage calls really needed?
499  addStage(consolidator);
500 
501  // --------------------------------------
502  // Optionally add a stage to the pipeline that drops all redundant result.
503  // The filter is applied here to remove redundancies early on directly after IDB and EDB
504  // results are combined.
505  // --------------------------------------
506  if (ctx->queryFlags & QUERY_FLAG_UNIQUE_SOLUTIONS) {
507  auto filterStage = std::make_shared<RedundantAnswerFilter>();
508  stepOutput >> filterStage;
509  lastOut = filterStage;
510  } else {
511  lastOut = stepOutput;
512  }
513  }
514 
515  lastOut >> pipelineOutput;
516 }

518 bool knowrob::EDBComparator::operator()(const FramedTriplePatternPtr &a, const FramedTriplePatternPtr &b) const {
519  // - prefer evaluation of literals with fewer variables
520  auto numVars_a = a->numVariables();
521  auto numVars_b = b->numVariables();
522  if (numVars_a != numVars_b) return (numVars_a > numVars_b);
523 
524  // - prefer literals with grounded predicate
525  bool hasProperty_a = (a->propertyTerm() && a->propertyTerm()->termType() == TermType::ATOMIC);
526  bool hasProperty_b = (b->propertyTerm() && b->propertyTerm()->termType() == TermType::ATOMIC);
527  if (hasProperty_a != hasProperty_b) return (hasProperty_a < hasProperty_b);
528 
529  // - prefer properties that appear less often in the EDB
530  if (hasProperty_a) {
531  auto numAsserts_a = vocabulary_->frequency(
532  std::static_pointer_cast<Atomic>(a->propertyTerm())->stringForm());
533  auto numAsserts_b = vocabulary_->frequency(
534  std::static_pointer_cast<Atomic>(b->propertyTerm())->stringForm());
535  if (numAsserts_a != numAsserts_b) return (numAsserts_a > numAsserts_b);
536  }
537 
538  return (a < b);
539 }
