21 void ConjunctiveBroadcaster::push(Channel &channel, const TokenPtr &tok) {
22  if (tok->tokenType() == TokenType::ANSWER_TOKEN) {
23  auto answer = std::static_pointer_cast<const Answer>(tok);
24  if (answer->isPositive()) {
25  const uint32_t channelID = channel.id();
26  // need to lock the whole push as genCombinations uses an iterator over the buffer.
27  std::lock_guard<std::mutex> lock(buffer_mutex_);
28 
29  // add to the buffer for later combinations
30  // replace other answer with same hash if present
31  buffer_[channelID][answer->hashOfAnswer()] = answer;
32 
33  // generate combinations with other channels if each channel
34  // buffer has some content.
35  if (buffer_.size() == channels_.size()) {
36  if (channels_.size() == 1) {
37  // not needed to generate combinations
38  TokenBroadcaster::push(tok);
39  } else {
40  // generate all combinations and push combined messages
41  genCombinations(channelID, buffer_.begin(), answer);
42  }
43  }
44  } else if (answer->isNegative()) {
45  // do not combine negative answers like the positive ones above.
46  // only push "no" when receiving EOF while no positive answer has been produced.
47  negativeAnswers_.emplace_back(std::static_pointer_cast<const AnswerNo>(answer));
48  }
49  } else {
50  if (tok->indicatesEndOfEvaluation() && !hasSolution_) {
51  if (negativeAnswers_.size() == 1) {
52  TokenBroadcaster::push(negativeAnswers_.front());
53  } else {
54  auto no = std::make_shared<AnswerNo>();
55  for (auto &x: negativeAnswers_) {
56  no->mergeWith(*x);
57  }
58  TokenBroadcaster::push(no);
59  }
60  }
61  // pass through non-answer messages
62  TokenStream::push(channel, tok);
63  }

66 void ConjunctiveBroadcaster::genCombinations( //NOLINT
67  uint32_t pushedChannelID, AnswerMap::iterator it, AnswerPtr &combinedResult) {
68  if (it == buffer_.end()) {
69  // end reached, push combination
70  TokenBroadcaster::push(combinedResult);
71  hasSolution_ = true;
72  } else if (it->first == pushedChannelID) {
73  // pass through channel from which the new message was pushed
74  auto it1 = it;
75  ++it1;
76  genCombinations(pushedChannelID, it1, combinedResult);
77  } else if (it->second.size() == 1) {
78  // only a single message buffered from this channel
79  auto it1 = it;
80  ++it1;
81  auto merged = mergeAnswers(combinedResult,
82  it->second.begin()->second, ignoreInconsistentAnswers_);
83  if (merged) {
84  genCombinations(pushedChannelID, it1, merged);
85  }
86  } else {
87  // generate a combination for each buffered message
88  // note: the number of possible combinations grows exponentially with number of messages in channels
89  auto it1 = it;

92  auto merged = mergeAnswers(
