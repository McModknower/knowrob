24 knowrob::IRIAtomPtr Property::reifiedIRI(std::string_view iri) {
25  // split the IRI at the last '#' and insert 'Reified' in between
26  auto pos = iri.rfind('#');
27  char delimiter = '#';
28  if (pos == std::string::npos) {
29  pos = iri.rfind('/');
30  delimiter = '/';
31  }
32  std::stringstream ss;
33  if (pos == std::string::npos) {
34  ss << "Reified_" << iri;
35  return IRIAtom::Tabled(ss.str());
36  }
37  ss << iri.substr(0, pos) << delimiter << "Reified_" << iri.substr(pos+1);
38  return IRIAtom::Tabled(ss.str());
39 }

41 knowrob::IRIAtomPtr Property::unReifiedIRI(std::string_view iri) {

83 void Property::forallParents(const PropertyVisitor &visitor,
84  bool includeSelf,
85  bool skipDuplicates) {
86  std::queue<Property *> queue_;
87  std::set<std::string_view> visited_;
88 
89  // push initial elements to the queue
90  if (includeSelf) queue_.push(this);
91  else for (auto &x: directParents_) queue_.push(x.get());
92 
93  // visit each parent
94  while (!queue_.empty()) {
95  auto front = queue_.front();
96  queue_.pop();
97  // visit popped property
98  visitor(*front);
99  // remember visited nodes
100  if (skipDuplicates) visited_.insert(front->iri());
101  // push parents of visited property on the queue
102  for (auto &directParent: front->directParents_) {
103  if (skipDuplicates && visited_.count(directParent->iri()) > 0) continue;
104  queue_.push(directParent.get());
105  }
106  }
107 }
