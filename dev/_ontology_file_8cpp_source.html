21 bool OntologyFile::load(const TripleHandler &callback) {
22  auto resolved = URI::resolve(uri());
23  auto newVersion = DataSource::getVersionFromURI(resolved);
24 
25  // some OWL files are downloaded compile-time via CMake,
26  // they are downloaded into owl/external e.g. there are SOMA.owl and DUL.owl.
27  auto download_path = std::filesystem::path("owl") / "external" /
28  std::filesystem::path(resolved).filename();
29  auto p = std::filesystem::path(URI::resolve(download_path.u8string()));
30  const std::string *importURI;
31  if (exists(p)) {
32  KB_DEBUG("Using downloaded ontology at '{}'.", p.u8string());
33  importURI = &p.native();
34  } else {
35  importURI = &resolved;
36  }
37 
38  KB_INFO("Loading ontology at '{}' with version "
39  "\"{}\" and origin \"{}\".", *importURI, newVersion, origin_);
40 
41  OntologyParser parser(*importURI, tripleFormat());
42  parser.setOrigin(origin_);
43  parser.setFrame(frame());
44  // filter is called for each triple, if it returns false, the triple is skipped
45  parser.setFilter([this](const FramedTriple &triple) {
46  return !vocabulary_->isAnnotationProperty(triple.predicate());
47  });
48  // define a prefix for naming blank nodes
49  parser.setBlankPrefix(std::string("_") + origin_);
50  auto result = parser.run([&callback](const TripleContainerPtr &triples) {
51  callback(triples);
52  });
53  if (!result) {
54  KB_WARN("Failed to parse ontology {} ({})", *importURI, uri());
55  return false;
56  }
57  // remember owl:imports
58  setImports(parser.imports());
59 
60  return true;
61 }
